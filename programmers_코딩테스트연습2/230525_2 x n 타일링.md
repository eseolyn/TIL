function solution(n) {
    let memo = new Array(n+1);
    memo[1] = 1;
    memo[2] = 2;
    
    let fibo = (n) => {
        if (memo[n] !== undefined){ 
            return memo[n];
        }
        memo[n] = (fibo(n - 1) + fibo(n - 2))%1000000007;
        return memo[n];
    };
    return fibo(n);
}
